1. Create a Campaign

Start by creating a fuzzing campaign to organize your testing efforts.

POST /api/v1/campaigns
{
  "name": "My Campaign",
  "target": "my-binary",
  "fuzzer": "afl++"
}

2. Submit a Job

Create fuzzing jobs within your campaign to start testing.

POST /api/v1/jobs
{
  "campaign_id": "{{campaign_id}}",
  "name": "Test Job",
  "duration": 3600
}

3. Monitor Progress

Track coverage, crashes, and performance metrics in real-time.

GET /api/v1/campaigns/{{id}}/stats
GET /api/v1/analytics/coverage-trend

API Key Authentication

All API requests require authentication using an API key. Include your API key in the request header:

X-API-Key: your-api-key-here

Example Request

curl -H "X-API-Key: your-api-key-here" \
     -H "Content-Type: application/json" \
     https://api.fuzztesting.com/v3/campaigns

Python Example

import requests

class fuzztestingClient:
    def __init__(self, api_key, base_url="http://localhost:8080/api/v1"):
        self.api_key = api_key
        self.base_url = base_url
        self.headers = {
            "X-API-Key": api_key,
            "Content-Type": "application/json"
        }
    
    def create_campaign(self, name, target, fuzzer="afl++"):
        data = {
            "name": name,
            "target": target,
            "fuzzer": fuzzer
        }
        response = requests.post(
            f"{self.base_url}/campaigns",
            json=data,
            headers=self.headers
        )
        return response.json()
    
    def get_campaign_stats(self, campaign_id):
        response = requests.get(
            f"{self.base_url}/campaigns/{campaign_id}/stats",
            headers=self.headers
        )
        return response.json()

# Usage
client = fuzztestingClient("your-api-key")
campaign = client.create_campaign("Security Test", "/bin/target")
stats = client.get_campaign_stats(campaign["id"])

Go Example

package main

import (
    "bytes"
    "encoding/json"
    "fmt"
    "net/http"
)

type fuzztestingClient struct {
    APIKey  string
    BaseURL string
}

func (c *fuzztestingClient) CreateJob(campaignID, name string, duration int) error {
    payload := map[string]interface{}{
        "campaign_id": campaignID,
        "name":        name,
        "duration":    duration,
    }
    
    data, _ := json.Marshal(payload)
    req, err := http.NewRequest("POST", c.BaseURL+"/jobs", bytes.NewBuffer(data))
    if err != nil {
        return err
    }
    
    req.Header.Set("X-API-Key", c.APIKey)
    req.Header.Set("Content-Type", "application/json")
    
    client := &http.Client{}
    resp, err := client.Do(req)
    if err != nil {
        return err
    }
    defer resp.Body.Close()
    
    if resp.StatusCode != http.StatusCreated {
        return fmt.Errorf("unexpected status: %d", resp.StatusCode)
    }
    
    return nil
}
